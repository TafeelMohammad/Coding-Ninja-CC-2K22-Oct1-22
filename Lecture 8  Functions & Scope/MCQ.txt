public static void main (String[] args) {
    int n=10,r=6;
    int factn=1,factr=1,factnr=1;

    {
        factn*=i;
        if(i<=r)
            factr*=i;
        if(i<=n-r)
            factnr*=i;
    }
    int ncr=factn/(factr*factnr);
    System.out.print(ncr);
}

210
